ACD_Turn'] = sigmoid(-df['M
ACD Hist'] * 0.8)

df['20d Vol %']> 1.6, 1.0,
np.clip((df['20d Vol %'] - 0.8) / 0.8, 0, 1)
)
# Final score
df['Rebound_Score'] = (
weights['rsi'] * df['RSI_Score'] +
weights['stoch'] * df['Stoch_Score'] +
weights['bb'] * df['BB_Score'] +
weights['vol_spike'] * df['Vol_Spike_Z'] +
weights['drawdown'] * df['Drawdown_Score'] +

import pandas as pd

from scipy import stats
class SignalEngine:
def __init__(self):
self.default_weights = {
'rsi': -0.26, 'stoch': -0.21, 'bb': -0.16, 'drawdown': -0.19,
'vol_spike': 0.11, 'macd': 0.07, 'vol20': 0.05, 'interaction': 0.12
}
def compute_signals(self, df: pd.DataFrame, custom_weights=None) -> pd.DataFrame:
weights = custom_weights or self.default_weights
df = df.copy()
# Z-score normalization across universe
for col in ['RSI', 'Stochastic', 'BB %', '20d Change %',
'Vol Spike', 'M
ACD Hist', '20d Vol %']:
df[f'z_{col}'] = stats.zscore(df[col].astype(float), nan_policy='omit')
# Core linear combination
df['Rebound_Score'] = (
weights['rsi'] * df['z_RSI'] +
weights['stoch'] * df['z_Stochastic'] +
weights['bb'] * df['z_BB %'] +
weights['drawdown'] * df['z_20d Change %'] +
weights['vol_spike'] * df['z_Vol Spike'] +
weights['macd'] * df['z_M
ACD Hist'] +
weights['vol20'] * df['z_20d Vol %']
# Capitulation interaction term (most predictive factor)
df['Rebound_Score'] += weights['interaction'] * (
-df['20d Change %'] * df['Vol Spike']
)
)
# Scale to 0-100 for interpretability
df['Rebound_Score'] = 50 + 10 * df['Rebound_Score']
df['Rebound_Score'] = df['Rebound_Score'].clip(0, 100)# N

import numpy as np
import pandas as pd
class SignalEngine:
def compute_signals(self, df: pd.DataFrame, weights: dict = None) -> pd.DataFrame:
if weights is None:

'drawdown': 0.10, 'macd': 0.10, 'vol_z': 0.05}
# Normalize oversold components (0-1 scale)
df['oversold_rsi'] = np.clip((50 - df['RSI']) / 50, 0, 1)
df['oversold_stoch'] = np.clip((80 - df['Stochastic']) / 80, 0, 1)
df['oversold_bb'] = np.clip((1 - df['BB %']) / 1, 0, 1)
df['vol_spike_norm'] = np.clip(df['Vol Spike'] / 2, 0, 1)
df['drawdown_norm'] = np.clip(-df['20d Change %'] / 30, 0, 1)
df['macd_bear_div'] = np.clip(-df['M
ACD Hist'] / 5, 0, 1)
# Vol z-score (simplified: assume 20d Vol % >2 is high; replace with rolling z if hist

# Confluence: count oversold (RSI/Stoch/BB
<20th percentile equiv: <20)
df['num_oversold'] = ((df['RSI'] < 20) & (df['Stochastic'] < 20) & (df['BB %'] < 0.2)).
oversold_mask = (df['RSI'] < 20) | (df['Stochastic'] < 20) | (df['BB %'] < 0.2)
df['num_oversold'] = oversold_mask.sum(axis=1) # Fix: per row sum
df['confluence_mult'] = 1 + 0.3 * np.clip(df['num_oversold'], 0, 3)
# Weighted sum * mult * 100
df['Rebound Score'] = 100 * (
weights['rsi'] * df['oversold_rsi'] +
weights['stoch'] * df['oversold_stoch'] +
weights['bb'] * df['oversold_bb'] +
weights['vol_spike'] * df['vol_spike_norm'] +
weights['drawdown'] * df['drawdown_norm'] +
weights['macd'] * df['macd_bear_div'] +
weights['vol_z'] * df['vol_z_oversold']
) * df['confluence_mult']
# Signal Score: Rebound * vol_spike (prioritize liquid)
df['Signal Score'] = df['Rebound Score'] * df['Vol Spike']
return df.sort_values('Rebound Score', ascending=False)

import pandas as pd
import numpy as np
from scipy.stats import zscore
class SignalEngine:
def compute_signals(self, df: pd.DataFrame, weights: dict = None) -> pd.DataFrame:
if weights is None:
weights = {'rsi': 0.22, 'stoch': 0.19, 'bb': 0.16, 'dd': 0.15,
'vol_spike': 0.12, 'macd': 0.09, 'vol_z': 0.07}
df = df.copy()
# Normalized components (higher = better for rebound)
df['rsi_norm'] = (80 - df['RSI'].clip(20, 80)) / 60
df['stoch_norm'] = (85 - df['Stochastic'].clip(10, 85)) / 75
df['bb_norm'] = (30 - df['BB %'].clip(-20, 30)) / 50
df['dd_norm'] = (-df['10d Change %'].clip(-45, 0)) / 45
df['vol_spike_norm'] = (df['Vol Spike'] - 0.6).clip(0, 4) / 3.0
df['macd_turn'] = np.tanh(df['M
ACD Hist'] + 1.2)
# Volatility z-score (20d vol vs trailing 60d)
if '20d Vol %' in df.columns:
df['vol_z'] = zscore(df['20d Vol %'].fillna(df['20d Vol %'].mean()))
df['vol_z_score'] = df['vol_z'].clip(0, 3) / 3.0
else:
df['vol_z_score'] = 0.5
# Core weighted score
df['Rebound Score'] = (
weights['rsi'] * df['rsi_norm'] +
weights['stoch'] * df['stoch_norm'] +
weights['bb'] * df['bb_norm'] +
weights['dd'] * df['dd_norm'] +
weights['vol_spike'] * df['vol_spike_norm'] +
weights['macd'] * df['macd_turn'] +
weights['vol_z'] * df['vol_z_score']
) * 100
# Confirmation multiplier (volume + liquidity filter)
df['Confirmation'] = 1 + 0.35 * (df['Vol Spike']> 1.1).astype(int)
df['Signal Score'] = df['Rebound Score'] * df['Confirmation']
# Final ranking
df = df.sort_values('Signal Score', ascending=False).reset_index(drop=True)
df['Rank'] = df.index + 1
return df

def compute_signals(self, df: pd.DataFrame, weights: dict = None) -> pd.DataFrame:

df = df.copy()
# Normalized component scores (0-100)
df['RSI_Score'] = np.clip((45 - df['RSI']) * 3.33, 0, 100)

df['BB_Score'] = np.clip((18 - df['BB %']) * 6.67, 0, 100)
df['DD_Score'] = np.clip(-df['10d Change %'] * 3.5, 0, 100)

ACD_Score'] = np.where(
df['M
ACD Hist'] < 0,
np.clip(-df['M
ACD Hist'] * 18, 0, 100),
np.clip(df['M
ACD Hist'] * 8, 0, 40) # decaying reward for positive
)

if '20d Vol %' in df.columns:
df['VolZ'] = (df['20d Vol %'] - 1.0) / 0.45 # 0.45 is approx historical std

df['Signal Score'] = df['Rebound Score'] * (1 + df['Vol Spike'] * 0.4) # volume boost
