# ====================== REQUEST HANDLERS ======================
class ContainerHandler(http.server.BaseHTTPRequestHandler):
    def do_GET(self):
        if self.path in ("/", "/index.html"):
            self.send_html_page()
        elif self.path == "/data":
            self.send_container_data()
        else:
            self.send_error(404, "Not Found")

    def do_POST(self):
        if self.path == "/grok":
            self.handle_grok_request()
        else:
            self.send_error(405, "Method Not Allowed")

    def send_html_page(self):
        self.send_response(200)
        self.send_header("Content-type", "text/html; charset=utf-8")
        self.end_headers()
        self.wfile.write(HTML_CONTENT.encode("utf-8"))

    def send_container_data(self):
        global VESSELS, PORTS, MANUFACTURERS

        # Realistic simulation (optimised)
        for vessel in VESSELS:
            lat_diff = vessel["dest_lat"] - vessel["lat"]
            lng_diff = vessel["dest_lng"] - vessel["lng"]
            vessel["lat"] += lat_diff * (0.028 + random.random() * 0.022)
            vessel["lng"] += lng_diff * (0.028 + random.random() * 0.022)
            vessel["course"] = (vessel["course"] + random.randint(-18, 18)) % 360
            vessel["speed_knots"] = max(8 if vessel["vessel_type"] == "Oil Tanker" else 12, min(23, vessel["speed_knots"] + random.uniform(-2.8, 3.2)))
            vessel["impact"] = max(12, min(94, vessel["impact"] + random.randint(-16, 18)))

        for p in PORTS:
            p["congestion"] = max(15, min(92, p["congestion"] + random.randint(-12, 15)))

        for vessel in VESSELS:
            for p in PORTS:
                if p["name"] == vessel["destination"]:
                    vessel["destination_congestion"] = p["congestion"]
                    break

        for m in MANUFACTURERS:
            m["delay_days"] = max(1, min(25, m["delay_days"] + random.randint(-3, 6)))
            m["risk_score"] = max(40, min(98, m["risk_score"] + random.randint(-8, 12)))

        payload = {"vessels": VESSELS, "ports": PORTS, "manufacturers": MANUFACTURERS}

        self.send_response(200)
        self.send_header("Content-type", "application/json")
        self.send_header("Access-Control-Allow-Origin", "*")
        self.end_headers()
        self.wfile.write(json.dumps(payload).encode("utf-8"))

    def handle_grok_request(self):
        try:
            content_length = int(self.headers.get('Content-Length', 0))
            post_data = self.rfile.read(content_length)
            data = json.loads(post_data.decode('utf-8'))

            user_message = data.get("message", "")
            client_history = data.get("history", [])

            if not GROK_API_KEY:
                reply = "❌ Grok API key not configured. Please check .streamlit/secrets.toml"
            else:
                messages = [{"role": entry["role"], "content": entry["content"]} for entry in client_history]
                messages.append({"role": "user", "content": user_message})

                api_payload = {
                    "model": "grok-beta",
                    "messages": messages,
                    "temperature": 0.7,
                    "max_tokens": 1200
                }

                api_resp = requests.post(
                    "https://api.x.ai/v1/chat/completions",
                    headers={
                        "Authorization": f"Bearer {GROK_API_KEY}",
                        "Content-Type": "application/json"
                    },
                    json=api_payload,
                    timeout=45
                )
                api_resp.raise_for_status()
                reply = api_resp.json()["choices"][0]["message"]["content"]

            # Permanent logging for self-improvement
            log_entry = {
                "timestamp": datetime.datetime.now().isoformat(),
                "user_message": user_message,
                "grok_reply": reply,
                "history_length": len(client_history)
            }
            with open(CHAT_LOG_FILE, "a") as f:
                f.write(json.dumps(log_entry) + "\n")

            self.send_response(200)
            self.send_header("Content-type", "application/json")
            self.send_header("Access-Control-Allow-Origin", "*")
            self.end_headers()
            self.wfile.write(json.dumps({"reply": reply}).encode("utf-8"))

        except Exception as e:
            self.send_response(500)
            self.send_header("Content-type", "application/json")
            self.send_header("Access-Control-Allow-Origin", "*")
            self.end_headers()
            self.wfile.write(json.dumps({"reply": f"Server error: {str(e)}"}).encode("utf-8"))
