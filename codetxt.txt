import subprocess
import json
from datetime import datetime
from pathlib import Path
CURRENT_VERSION = "4.1"
VERSION_HISTORY_FILE = Path("self_improvement_history.json")
def get_current_source() -> str:
    """Read our own source code for 100% accurate context."""
    try:
        return Path(__file__).read_text(encoding="utf-8")
    except Exception as e:
        return f"# Error reading source: {e}"
def load_version_history() -> list:
    if VERSION_HISTORY_FILE.exists():
        return json.loads(VERSION_HISTORY_FILE.read_text(encoding="utf-8"))
    return []
def record_version_bump(new_version: str, summary: str, previous_version: str = None):
    history = load_version_history()
    history.append({
        "version": new_version,
        "timestamp": datetime.utcnow().isoformat(),
        "previous_version": previous_version or (history[-1]["version"] if history else "0.0"),
        "summary": summary,
    })
    VERSION_HISTORY_FILE.write_text(json.dumps(history, indent=2), encoding="utf-8")
    print(f"✓ Version bump recorded: v{previous_version or '4.0'} → v{new_version}")
def apply_patch(patch_content: str, auto_apply: bool = False) -> bool:
    """Auto-patch mode with user confirmation."""
    patch_file = Path("update_v4.1.patch")
    patch_file.write_text(patch_content, encoding="utf-8")
    
    print("\n=== PROPOSED PATCH ===")
    print(patch_content)
    print("=====================\n")
    
    if not auto_apply:
        confirm = input("Apply this patch using git apply? (y/N): ").strip().lower()
        if confirm not in ('y', 'yes'):
            print("Patch cancelled by user.")
            return False
    
    try:
        result = subprocess.run(["git", "apply", "--check", str(patch_file)], 
                              capture_output=True, text=True)
        if result.returncode != 0:
            print("Patch check failed:", result.stderr)
            return False
            
        subprocess.run(["git", "apply", str(patch_file)], check=True)
        print("✅ Patch successfully applied!")
        record_version_bump(CURRENT_VERSION, "Implemented auto-patch mode, self-source injection, and version lineage")
        return True
    except subprocess.CalledProcessError as e:
        print("Failed to apply patch:", e)
        return False
